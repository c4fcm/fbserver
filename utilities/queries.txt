== Get a list of how many corrections a given set of users has carried out: ==

select u.screen_name, count(*) as c from account_gender_judgments as a JOIN users as u on a.user_id = u.id GROUP BY u.screen_name HAVING c > 0 ORDER BY c DESC;

== Get a list of people who took the survey more than a week ago ==
select screen_name, treatment, updated_at from users where survey_complete=true AND updated_at <= DATE_SUB(NOW(), INTERVAL 7 DAY) ORDER BY treatment;
